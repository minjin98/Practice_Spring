
ALTER SESSION SET "_ORACLE_SCRIPT"=true;

-- USER SQL
CREATE USER "HELLOUSER" IDENTIFIED BY "HELLOUSER"  
DEFAULT TABLESPACE "SOLDB"
TEMPORARY TABLESPACE "TEMP";

-- QUOTAS
ALTER USER "HELLOUSER" QUOTA UNLIMITED ON "SOLDB";

-- ROLES
GRANT "CONNECT" TO "HELLOUSER" ;
GRANT "RESOURCE" TO "HELLOUSER" ;

-- SYSTEM PRIVILEGES
GRANT CREATE TRIGGER TO "HELLOUSER" ;
GRANT CREATE VIEW TO "HELLOUSER" ;
GRANT CREATE SESSION TO "HELLOUSER" ;
GRANT CREATE TABLE TO "HELLOUSER" ;
GRANT CREATE SYNONYM TO "HELLOUSER" ;
GRANT CREATE SEQUENCE TO "HELLOUSER" ;
